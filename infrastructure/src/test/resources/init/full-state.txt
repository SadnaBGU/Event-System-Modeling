# ── Auth ─────────────────────────────────────────────────────────────────────
login(founder, password123)

# ── Company A: full event/lottery/policy/purchase flow ───────────────────────
open-production-company(founder, acme, "Acme Live", "Top live concerts", 4.5)

# Appoint a manager, have them accept, then widen their permissions
appoint-manager(founder, acme, manager, EVENT_INVENTORY_MANAGEMENT|MODIFY_POLICIES)
accept-appointment(manager, acme)
modify-manager-permissions(founder, acme, manager, EVENT_INVENTORY_MANAGEMENT|MODIFY_POLICIES|VIEW_PURCHASE_HISTORY)

# Appoint a co-owner and have them accept
appoint-owner(founder, acme, coowner)
accept-appointment(coowner, acme)

# Event with one standing and one seated zone
create-event(founder, acme, summerfest, "Summer Fest", 2026-09-01T20:00, Festival, "Tel Aviv Park", "Annual summer festival")
create-standing-zone(founder, summerfest, ga, "General Admission", 150.00, ILS, 500)
create-seated-zone(founder, summerfest, vip, "VIP", 450.00, ILS, A:10|B:10)

# Company purchase & discount policies
set-purchase-policy(founder, acme, COMPANY, COMPANY_WIDE, "Max 5 tickets", true, MAX_TICKETS, 5)
set-discount-policy(founder, acme, COMPANY, COMPANY_WIDE, "Visible Discount", false, true, "Early Bird", 15, ALLOW_ALL, -, VISIBLE)

# Publish so it is purchasable
publish-event(founder, summerfest)

# Lottery: open, two entrants, draw one winner
open-lottery(founder, summerfest, lottery1)
register-lottery(manager, lottery1)
register-lottery(buyer, lottery1)
draw-lottery(founder, lottery1, 1)

# Reserve standing tickets and check out (payment + issuance mocked to succeed)
reserve(buyer, order1, ga, 2)
checkout(order1, tok_visa_mock, )
